91/**
92 * @brief Configures a GPIO pin as a push-pull output.
93 *
94 * @details
95 * Sets the pin to output mode with no pull resistors and applies the initial
96 * logic level.
97 *
98 * Preconditions:
99 * - pin must be a valid output-capable GPIO.
100 *
101 * Postconditions:
102 * - GPIO is configured for output and driven to initial_level.
103 *
104 * Side effects:
105 * - Updates GPIO hardware configuration.
106 *
107 * @param pin GPIO number to configure.
108 * @param initial_level Initial logic level to drive.
109 * @return HAL_OK on success, HAL_ERR_INVALID on invalid pin/config failure.
110 */
111hal_status_t hal_gpio_config_output(int pin, hal_gpio_level_t initial_level) {
112 if (!hal_gpio_output_valid(pin)) {
113 return HAL_ERR_INVALID;
114 }
115
116 gpio_config_t cfg = {
117 .pin_bit_mask = 1ULL << pin,
118 .mode = GPIO_MODE_OUTPUT,
119 .pull_up_en = GPIO_PULLUP_DISABLE,
120 .pull_down_en = GPIO_PULLDOWN_DISABLE,
121 .intr_type = GPIO_INTR_DISABLE,
122 };
123 if (gpio_config(&cfg) != ESP_OK) {
124 return HAL_ERR_INVALID;
125 }
126 gpio_set_level((gpio_num_t)pin, (int)initial_level);
127 return HAL_OK;
128}

213typedef struct {

219static hal_timer_entry_t hal_timers[HAL_TIMER_MAX];

221static void hal_timer_dispatch(void *arg) {
222 hal_timer_entry_t *entry = (hal_timer_entry_t *)arg;
223 if (entry->callback) {
224 entry->callback(entry->user_data);
225 }

228/**
229 * @brief Starts a periodic HAL timer.
230 *
231 * @details
232 * Allocates or reuses a timer slot and calls the provided callback at the
233 * specified period.
234 *
235 * Preconditions:
236 * - timer_id must be in range [0, HAL_TIMER_MAX).
237 * - callback must be non-null.
238 * - period_ms must be > 0.
239 *
240 * Postconditions:
241 * - The timer slot is active and invokes callback periodically.
242 *
243 * Side effects:
244 * - Allocates an ESP timer and schedules periodic callbacks.
245 *
246 * @param timer_id Timer slot identifier.
247 * @param period_ms Period in milliseconds.
248 * @param callback Function to invoke each period. Must not be NULL.
249 * @param user_data Opaque pointer passed to callback.
250 * @return HAL_OK on success, HAL_ERR_INVALID on invalid parameters or failure.
251 */
252hal_status_t hal_timer_start(int timer_id, uint32_t period_ms,
253 hal_timer_callback_t callback, void *user_data) {
254 if (timer_id < 0 || timer_id >= HAL_TIMER_MAX || callback == NULL || period_ms == 0) {
255 return HAL_ERR_INVALID;
256 }
257
258 hal_timer_entry_t *entry = &hal_timers[timer_id];
259 if (entry->handle) {
260 esp_timer_stop(entry->handle);
261 esp_timer_delete(entry->handle);
262 entry->handle = NULL;
263 }
264
265 esp_timer_create_args_t args = {
266 .callback = &hal_timer_dispatch,
267 .arg = entry,
268 .name = "hal_timer"
269 };
270 if (esp_timer_create(&args, &entry->handle) != ESP_OK) {
271 entry->handle = NULL;
272 return HAL_ERR_INVALID;
273 }
274
275 entry->callback = callback;
276 entry->user_data = user_data;
277
278 uint64_t period_us = (uint64_t)period_ms * 1000ULL;
279 if (esp_timer_start_periodic(entry->handle, period_us) != ESP_OK) {
280 esp_timer_delete(entry->handle);
281 entry->handle = NULL;
282 return HAL_ERR_INVALID;
283 }
284
285 return HAL_OK;
286}

288/**
289 * @brief Stops a periodic HAL timer.
290 *
291 * @details
292 * Cancels and deletes the timer slot if it is active.
293 *
294 * Preconditions:
295 * - timer_id must be in range [0, HAL_TIMER_MAX).
296 *
297 * Postconditions:
298 * - The timer slot is inactive.
299 *
300 * Side effects:
301 * - Deletes the underlying ESP timer if present.
302 *
303 * @param timer_id Timer slot identifier.
304 * @return HAL_OK on success, HAL_ERR_INVALID on invalid timer_id.
305 */
306hal_status_t hal_timer_stop(int timer_id) {
307 if (timer_id < 0 || timer_id >= HAL_TIMER_MAX) {
308 return HAL_ERR_INVALID;
309 }
310
311 hal_timer_entry_t *entry = &hal_timers[timer_id];
312 if (!entry->handle) {
313 return HAL_OK;
314 }
315
316 esp_timer_stop(entry->handle);
317 esp_timer_delete(entry->handle);
318 entry->handle = NULL;
319 entry->callback = NULL;
320 entry->user_data = NULL;
321 return HAL_OK;
322}

335/**
336 * @brief Starts the global HAL tick timer.
337 *
338 * @details
339 * Configures a periodic ESP timer that calls the provided callback at the
340 * requested frequency.
341 *
342 * Preconditions:
343 * - hz must be > 0.
344 * - callback must be non-null.
345 *
346 * Postconditions:
347 * - The tick callback is invoked at approximately hz.
348 *
349 * Side effects:
350 * - Allocates and starts an ESP timer.
351 *
352 * @param hz Tick frequency in Hertz.
353 * @param callback Function to invoke on each tick. Must not be NULL.
354 * @param user_data Opaque pointer passed to callback.
355 * @return HAL_OK on success, HAL_ERR_INVALID on invalid params or failure.
356 */
357hal_status_t hal_tick_start(uint32_t hz, hal_timer_callback_t callback, void *user_data) {
358 if (hz == 0 || callback == NULL) {
359 return HAL_ERR_INVALID;
360 }
361
362 if (hal_tick_timer) {
363 esp_timer_stop(hal_tick_timer);
364 esp_timer_delete(hal_tick_timer);
365 hal_tick_timer = NULL;
366 }
367
368 uint64_t period_us = 1000000ULL / hz;
369 if (period_us == 0) {
370 return HAL_ERR_INVALID;
371 }
372
373 esp_timer_create_args_t args = {
374 .callback = &hal_tick_dispatch,
375 .arg = NULL,
376 .name = "hal_tick"
377 };
378 if (esp_timer_create(&args, &hal_tick_timer) != ESP_OK) {
379 hal_tick_timer = NULL;
380 return HAL_ERR_INVALID;
381 }
382
383 hal_tick_callback = callback;
384 hal_tick_user_data = user_data;
385
386 if (esp_timer_start_periodic(hal_tick_timer, period_us) != ESP_OK) {
387 esp_timer_delete(hal_tick_timer);
388 hal_tick_timer = NULL;
389 return HAL_ERR_INVALID;
390 }
391
392 return HAL_OK;
393}

429 if (bits < 1) {

438typedef struct {
439 bool configured;
440 int pin;
441 ledc_timer_t timer;
442 uint32_t duty_max;
443} hal_pwm_channel_t;
444
445static hal_pwm_channel_t hal_pwm_channels[LEDC_CHANNEL_MAX];
446static bool hal_pwm_timer_configured[LEDC_TIMER_MAX];
447static uint32_t hal_pwm_timer_freq[LEDC_TIMER_MAX];
448static uint32_t hal_pwm_timer_bits[LEDC_TIMER_MAX];

450/**
451 * @brief Initializes a PWM channel for a GPIO pin.
452 *
453 * @details
454 * Configures the LEDC timer and channel for the requested frequency and
455 * resolution. Motor pins are placed on a separate LEDC timer to preserve
456 * frequency settings.
457 *
458 * Preconditions:
459 * - channel must be in [0, LEDC_CHANNEL_MAX).
460 * - pin must be a valid output-capable GPIO.
461 *
462 * Postconditions:
463 * - The PWM channel is configured and ready for duty updates.
464 *
465 * Side effects:
466 * - Allocates/updates LEDC timer and channel configuration.
467 *
468 * @param channel LEDC channel index.
469 * @param pin GPIO number to drive with PWM.
470 * @param freq_hz PWM frequency in Hertz.
471 * @param duty_resolution_bits Duty resolution in bits.
472 * @return HAL_OK on success, HAL_ERR_INVALID on invalid params or failure.
473 */
474hal_status_t hal_pwm_init(int channel, int pin, uint32_t freq_hz,
475 uint32_t duty_resolution_bits) {
476 if (channel < 0 || channel >= LEDC_CHANNEL_MAX) {
477 return HAL_ERR_INVALID;
478 }
479 if (!hal_gpio_output_valid(pin)) {
480 return HAL_ERR_INVALID;
481 }
482
483 ledc_timer_t timer = hal_pwm_select_timer(pin);
484 ledc_timer_bit_t res = hal_pwm_bits_to_ledc(duty_resolution_bits);
485
486 if (!hal_pwm_timer_configured[timer] ||
487 hal_pwm_timer_freq[timer] != freq_hz ||
488 hal_pwm_timer_bits[timer] != duty_resolution_bits) {
489 ledc_timer_config_t timer_cfg = {
490 .speed_mode = LEDC_LOW_SPEED_MODE,
491 .timer_num = timer,
492 .duty_resolution = res,
493 .freq_hz = freq_hz,
494 .clk_cfg = LEDC_AUTO_CLK,
495 };
496 if (ledc_timer_config(&timer_cfg) != ESP_OK) {
497 return HAL_ERR_INVALID;
498 }
499 hal_pwm_timer_configured[timer] = true;
500 hal_pwm_timer_freq[timer] = freq_hz;
501 hal_pwm_timer_bits[timer] = duty_resolution_bits;
502 }
503
504 ledc_channel_config_t channel_cfg = {
505 .gpio_num = pin,
506 .speed_mode = LEDC_LOW_SPEED_MODE,
507 .channel = (ledc_channel_t)channel,
508 .timer_sel = timer,
509 .duty = 0,
510 .hpoint = 0,
511 };
512 if (ledc_channel_config(&channel_cfg) != ESP_OK) {
513 return HAL_ERR_INVALID;
514 }
515
516 hal_pwm_channels[channel].configured = true;
517 hal_pwm_channels[channel].pin = pin;
518 hal_pwm_channels[channel].timer = timer;
519 if (duty_resolution_bits >= 31) {
520 hal_pwm_channels[channel].duty_max = 0xFFFFFFFFU;
521 } else {
522 hal_pwm_channels[channel].duty_max = (1U << duty_resolution_bits) - 1U;
523 }
524
525 return HAL_OK;
526}

528/**
529 * @brief Updates the PWM duty for a configured channel.
530 *
531 * @details
532 * Clamps duty to the configured maximum for the channel before updating.
533 *
534 * Preconditions:
535 * - channel must be configured via hal_pwm_init().
536 *
537 * Side effects:

562 return HAL_ERR_INVALID;
563 }
564 return HAL_OK;

567/**
568 * @brief Initializes UART0 with the configured pins.
569 *
570 * @details
571 * Only UART0 is supported. Pins are defined in hal_config.h and must be set
572 * to non-BOARD_GPIO_UNUSED to enable UART usage.
573 *
574 * Preconditions:
575 * - uart_id must be UART_NUM_0.
576 * - UART pins must be configured in hal_config.h.
577 *
578 * Postconditions:
579 * - UART driver is installed and ready for IO.
580 *
581 * Side effects:
582 * - Allocates UART driver resources.
583 *
584 * @param uart_id UART identifier (must be UART_NUM_0).
585 * @param baud_rate Baud rate in bits per second.
586 * @return HAL_OK on success, HAL_ERR_UNSUPPORTED on unsupported config,
587 * HAL_ERR_INVALID on invalid parameters or driver failure.
588 */
589hal_status_t hal_uart_init(int uart_id, uint32_t baud_rate) {
590 if (uart_id != UART_NUM_0) {
591 return HAL_ERR_UNSUPPORTED;
592 }
593
594 if (HAL_UART0_TX_PIN == BOARD_GPIO_UNUSED || HAL_UART0_RX_PIN == BOARD_GPIO_UNUSED) {
595 return HAL_ERR_UNSUPPORTED;
596 }
597
598 if (!hal_gpio_valid(HAL_UART0_TX_PIN) || !hal_gpio_valid(HAL_UART0_RX_PIN)) {
599 return HAL_ERR_INVALID;
600 }
601
602 uart_config_t config = {
603 .baud_rate = (int)baud_rate,
604 .data_bits = UART_DATA_8_BITS,
605 .parity = UART_PARITY_DISABLE,
606 .stop_bits = UART_STOP_BITS_1,
607 .flow_ctrl = UART_HW_FLOWCTRL_DISABLE,
608 .source_clk = UART_SCLK_DEFAULT,
609 };
610
611 if (uart_param_config(uart_id, &config) != ESP_OK) {
612 return HAL_ERR_INVALID;
613 }
614
615 int rts = (HAL_UART0_RTS_PIN == BOARD_GPIO_UNUSED) ? UART_PIN_NO_CHANGE : HAL_UART0_RTS_PIN;
616 int cts = (HAL_UART0_CTS_PIN == BOARD_GPIO_UNUSED) ? UART_PIN_NO_CHANGE : HAL_UART0_CTS_PIN;
617 if (uart_set_pin(uart_id, HAL_UART0_TX_PIN, HAL_UART0_RX_PIN, rts, cts) != ESP_OK) {
618 return HAL_ERR_INVALID;
619 }
620
621 if (uart_driver_install(uart_id, 256, 0, 0, NULL, 0) != ESP_OK) {
622 return HAL_ERR_INVALID;
623 }
624
625 return HAL_OK;
626}

793/**
794 * @brief Initializes the SPI bus and device handle.
795 *
796 * @details
797 * Only bus_id 0 is supported and mapped to SPI2_HOST. Pins are defined in
798 * hal_config.h.
799 *
800 * Preconditions:
801 * - SPI pins must be configured and valid.
802 *
803 * Postconditions:
804 * - SPI bus is initialized and device handle is ready.
805 *
806 * Side effects:
807 * - Allocates SPI bus resources.
808 *
809 * @param bus_id SPI bus identifier (must be 0).
810 * @param clock_hz Clock rate in Hertz. Zero uses HAL_SPI0_CLOCK_HZ.
811 * @param mode SPI mode (0-3).
812 * @return HAL_OK on success, HAL_ERR_UNSUPPORTED on unsupported config,
813 * HAL_ERR_INVALID on invalid parameters or driver failure.
814 */
815hal_status_t hal_spi_init(int bus_id, uint32_t clock_hz, uint8_t mode) {
816 if (bus_id != 0) {
817 return HAL_ERR_UNSUPPORTED;
818 }
819 if (HAL_SPI0_MOSI_PIN == BOARD_GPIO_UNUSED || HAL_SPI0_MISO_PIN == BOARD_GPIO_UNUSED ||
820 HAL_SPI0_SCLK_PIN == BOARD_GPIO_UNUSED) {
821 return HAL_ERR_UNSUPPORTED;
822 }
823 if (!hal_gpio_valid(HAL_SPI0_MOSI_PIN) || !hal_gpio_valid(HAL_SPI0_MISO_PIN) ||
824 !hal_gpio_valid(HAL_SPI0_SCLK_PIN)) {
825 return HAL_ERR_INVALID;
826 }
827 if (hal_spi_initialized) {
828 return HAL_OK;
829 }
830
831 spi_bus_config_t buscfg = {
832 .mosi_io_num = HAL_SPI0_MOSI_PIN,
833 .miso_io_num = HAL_SPI0_MISO_PIN,
834 .sclk_io_num = HAL_SPI0_SCLK_PIN,
835 .quadwp_io_num = -1,
836 .quadhd_io_num = -1,
837 .max_transfer_sz = 4096,
838 };
839
840 if (spi_bus_initialize(SPI2_HOST, &buscfg, SPI_DMA_CH_AUTO) != ESP_OK) {
841 return HAL_ERR_INVALID;
842 }
843
844 spi_device_interface_config_t devcfg = {
845 .clock_speed_hz = (clock_hz == 0) ? HAL_SPI0_CLOCK_HZ : clock_hz,
846 .mode = mode,
847 .spics_io_num = (HAL_SPI0_CS_PIN == BOARD_GPIO_UNUSED) ? -1 : HAL_SPI0_CS_PIN,
848 .queue_size = 1,
849 };
850
851 if (spi_bus_add_device(SPI2_HOST, &devcfg, &hal_spi_device) != ESP_OK) {
852 return HAL_ERR_INVALID;
853 }
854
855 hal_spi_initialized = true;
856 return HAL_OK;
857}

967/**
968 * @brief Runs a basic HAL self-test sequence.
969 *
970 * @details
971 * Validates GPIO, tick timer, PWM, and optional peripheral bring-up. Only
972 * compiled when HAL_SELFTEST is defined.
973 *
974 * Side effects:
975 * - Reconfigures GPIO/PWM peripherals for test patterns.
976 * - Writes test output to the UART when available.
977 *
978 * @note Intended for development builds only.
979 */
980void hal_selftest_run(void) {
981#ifndef HAL_SELFTEST
982 return;
983#else
984 ESP_LOGI(TAG, "HAL self-test start");
985
986 int led_pins[] = {
987 BOARD_GPIO_LED0,
988 BOARD_GPIO_LED1,
989 BOARD_GPIO_LED2,
990 BOARD_GPIO_LED3
991 };
992
993 for (size_t i = 0; i < sizeof(led_pins) / sizeof(led_pins[0]); i++) {
994 if (hal_gpio_config_output(led_pins[i], HAL_GPIO_LOW) == HAL_OK) {
995 hal_gpio_write(led_pins[i], HAL_GPIO_HIGH);
996 hal_delay_ms(200);
997 hal_gpio_write(led_pins[i], HAL_GPIO_LOW);
998 }
999 }
1000
1001 hal_gpio_config_input(BOARD_GPIO_BUTTON_FWD, HAL_GPIO_PULL_UP);
1002 hal_gpio_config_input(BOARD_GPIO_BUTTON_REV, HAL_GPIO_PULL_UP);
1003 ESP_LOGI(TAG, "Button FWD=%d REV=%d",
1004 hal_gpio_read(BOARD_GPIO_BUTTON_FWD),
1005 hal_gpio_read(BOARD_GPIO_BUTTON_REV));
1006
1007 volatile uint32_t tick_count = 0;
1008 hal_tick_start(1000, hal_selftest_tick_cb, (void *)&tick_count);
1009 hal_delay_ms(1100);
1010 hal_tick_stop();
1011 ESP_LOGI(TAG, "Tick count (1s): %u", (unsigned)tick_count);
1012
1013 hal_pwm_init(0, BOARD_GPIO_LED0, HAL_PWM_LED_FREQ_HZ, HAL_PWM_DUTY_RES_BITS);
1014 hal_pwm_set_duty(0, (1U << (HAL_PWM_DUTY_RES_BITS - 1)));
1015
1016 if (hal_uart_init(UART_NUM_0, HAL_UART0_BAUD_DEFAULT) == HAL_OK) {
1017 const char banner[] = "HAL UART0 ready\n";
1018 hal_uart_write(UART_NUM_0, (const uint8_t *)banner, sizeof(banner) - 1);
1019 }
1020
1021 if (HAL_I2C0_SDA_PIN != BOARD_GPIO_UNUSED && HAL_I2C0_SCL_PIN != BOARD_GPIO_UNUSED) {
1022 hal_i2c_init(I2C_NUM_0, HAL_I2C0_CLOCK_HZ);
1023 }
1024
1025 if (HAL_SPI0_MOSI_PIN != BOARD_GPIO_UNUSED && HAL_SPI0_SCLK_PIN != BOARD_GPIO_UNUSED) {
1026 hal_spi_init(0, HAL_SPI0_CLOCK_HZ, HAL_SPI0_MODE);
1027 }
1028
1029 if (HAL_ADC_DEFAULT_CHANNEL != BOARD_GPIO_UNUSED) {
1030 int adc_value = hal_adc_read(HAL_ADC_DEFAULT_CHANNEL);
1031 ESP_LOGI(TAG, "ADC default channel=%d value=%d", HAL_ADC_DEFAULT_CHANNEL, adc_value);
1032 }
1033
1034 ESP_LOGI(TAG, "HAL self-test complete");
1035#endif
1036}
